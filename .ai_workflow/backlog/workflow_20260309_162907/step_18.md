protected async _notifyObservers(snapshot: T): Promise<void> {
  for (const callback of this._observers) {
    try {
      await Promise.resolve(callback(snapshot));
    } catch (err: unknown) {

    }

}

async notifyObservers(...args: T): Promise<void> {

      if (observer.update) await Promise.resolve(observer.update(...args));
    } catch (err) {
      // log error
    }
  }
}
async notifyFunctionObservers(...args: T): Promise<void> {
